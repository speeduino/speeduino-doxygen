1/*
2This is universal CAN library for STM32 that was made to be used with Speeduino EFI.
3It should support all STM32 MCUs that are also supported in stm32duino Arduino_Core_STM32 and supports up to 3x CAN buses.
4The library is created because at least currently (year 2021) there is no official CAN library in the STM32 core.
5This library is based on several STM32 CAN example libraries linked below and it has been combined with few
6things from Teensy FlexCAN library to make it compatible with the CAN features that exist in speeduino for Teensy.
7Links to repositories that have helped with this:
8https://github.com/nopnop2002/Arduino-STM32-CAN
9https://github.com/J-f-Jensen/libraries/tree/master/STM32_CAN
10https://github.com/jiauka/STM32F1_CAN
11
12STM32 core: https://github.com/stm32duino/Arduino_Core_STM32
13*/

23typedef struct CAN_message_t {
24 uint32_t id = 0; // can identifier
25 uint16_t timestamp = 0; // time when message arrived
26 uint8_t idhit = 0; // filter that id came from
27 struct {
28 bool extended = 0; // identifier is extended (29-bit)
29 bool remote = 0; // remote transmission request packet type
30 bool overrun = 0; // message overrun
31 bool reserved = 0;
32 } flags;
33 uint8_t len = 8; // length of data
34 uint8_t buf[8] = { 0 }; // data
35 int8_t mb = 0; // used to identify mailbox reception
36 uint8_t bus = 1; // used to identify where the message came (CAN1, CAN2 or CAN3)
37 bool seq = 0; // sequential frames
38} CAN_message_t;
