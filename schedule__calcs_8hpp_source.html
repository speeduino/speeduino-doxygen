9static inline uint16_t calculateInjectorStartAngle(uint16_t pwDegrees, int16_t injChannelDegrees, uint16_t injAngle)
10{
11 // 0<=injAngle<=720°
12 // 0<=injChannelDegrees<=720°
13 // 0<pwDegrees<=??? (could be many crank rotations in the worst case!)
14 // 45<=CRANK_ANGLE_MAX_INJ<=720
15 // (CRANK_ANGLE_MAX_INJ can be as small as 360/nCylinders. E.g. 45° for 8 cylinder)
16
17 uint16_t startAngle = (uint16_t)injAngle + (uint16_t)injChannelDegrees;
18
19 while (startAngle<pwDegrees) { startAngle = startAngle + (uint16_t)CRANK_ANGLE_MAX_INJ; } // Avoid underflow
20 startAngle = startAngle - pwDegrees; // startAngle guaranteed to be >=0.
21 while (startAngle>(uint16_t)CRANK_ANGLE_MAX_INJ) { startAngle = startAngle - (uint16_t)CRANK_ANGLE_MAX_INJ; } // Clamp to 0<=startAngle<=CRANK_ANGLE_MAX_INJ
22
23 return startAngle;
24}

41static inline __attribute__((always_inline)) uint32_t _calculateAngularTime(const Schedule &schedule, uint16_t angleOffset, uint16_t eventAngle, uint16_t crankAngle, uint16_t maxAngle) {
42 if (angleOffset==0U) { // Optimize for zero channel angle - no need to adjust start & crank angles
43 return _calculateAngularTime(schedule, eventAngle, crankAngle, maxAngle);
44 }
45 // Realign the current crank angle and the desired start angle around 0 degrees for the given cylinder/output
46 // Eg: If cylinder 2 TDC is 180 degrees after cylinder 1 (E.g. a standard 4 cylinder engine), then
47 // adjusted crank angle is 180* less than the current crank angle and adjusted start angle is the desired open angle less 180*.
48 // Thus the cylinder is being treated relative to its own TDC, regardless of its offset
49 //
50 // This is done to avoid very small or very large deltas between crank angle and start angle.
51 return _calculateAngularTime(schedule,
52 _adjustToTDC(eventAngle, angleOffset, maxAngle),
53 _adjustToTDC(crankAngle, angleOffset, maxAngle),
54 maxAngle);
55}

106// The concept here is that we have a more accurate crank angle.
107// Ignition timing is driven by target spark angle relative to crank position.
108// So the timing to begin & end charging the coil is based on crank angle.
109// With a more accurate crank angle, we can increase the precision of the
110// spark timing.
111static inline void adjustCrankAngle(IgnitionSchedule &schedule, int16_t crankAngle) {
112 constexpr uint8_t MIN_CYCLES_FOR_CORRECTION = 6U;
113
114 ATOMIC() { // Prevent race conditions with the timer interrupt.
115 // We only want to adjust the crank angle if we are running and the coil is charging or we are waiting for the timer to fire.
116 if( isRunning(schedule) ) {
117 if (schedule.dischargeAngle>crankAngle) {
118 // Coil is charging so change the charge time so the spark fires at
119 // the requested crank angle (this could reduce dwell time & potentially
120 // result in a weaker spark).
121 SET_COMPARE(schedule._compare, schedule._counter + angleToTimerTicks( ignitionLimits(schedule.dischargeAngle-crankAngle) ));
122 }
123 }
124 else if( (schedule.Status==PENDING) ) {
125 if ((currentStatus.startRevolutions > MIN_CYCLES_FOR_CORRECTION) && (schedule.chargeAngle>crankAngle)) {
126 // We are waiting for the timer to fire & start charging the coil.
127 // Keep dwell (I.e. duration) constant (for better spark) - instead adjust the waiting period so
128 // the spark fires at the requested crank angle.
129 SET_COMPARE(schedule._compare, schedule._counter + angleToTimerTicks( ignitionLimits(schedule.chargeAngle-crankAngle) ));
130 }
131 } else {
132 // Unknown state, so no adjustment possible
133 }
134 }
