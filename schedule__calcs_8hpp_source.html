7inline uint16_t calculateInjectorStartAngle(uint16_t pwDegrees, int16_t injChannelDegrees, uint16_t injAngle)
8{
9 // 0<=injAngle<=720°
10 // 0<=injChannelDegrees<=720°
11 // 0<pwDegrees<=??? (could be many crank rotations in the worst case!)
12 // 45<=CRANK_ANGLE_MAX_INJ<=720
13 // (CRANK_ANGLE_MAX_INJ can be as small as 360/nCylinders. E.g. 45° for 8 cylinder)
14
15 uint16_t startAngle = injAngle + injChannelDegrees;
16 // Avoid underflow
17 while (startAngle<pwDegrees) { startAngle = startAngle + CRANK_ANGLE_MAX_INJ; }
18 // Guarenteed to be >=0.
19 startAngle = startAngle - pwDegrees;
20 // Clamp to 0<=startAngle<=CRANK_ANGLE_MAX_INJ
21 while (startAngle>(uint16_t)CRANK_ANGLE_MAX_INJ) { startAngle = startAngle - CRANK_ANGLE_MAX_INJ; }
22
23 return startAngle;
24}
