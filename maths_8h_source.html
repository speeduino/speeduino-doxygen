223static inline uint32_t percentageApprox(uint16_t percent, uint32_t value) {
224 // To keep the percentage within 16-bits (for performance), we have to scale the precision based on the percentage.
225 // I.e. the larger the percentage, the smaller the precision has to be (and vice-versa).
226 //
227 // We could use __builtin_clz() and use the leading zero count as the precision, but that is slow:
228 // * AVR doesn't have a clz ASM instruction, so __builtin_clz() is implemented in software
229 // * It would require removing some compile time optimizations
230 #define TEST_AND_APPLY(precision) \
231 if (percent<(UINT16_C(1)<<(UINT16_C(16)-(precision)))) { \
232 return _percentageApprox<(precision)>(percent, value); \
233 }
234
235 TEST_AND_APPLY(9) // Percent<128
236 TEST_AND_APPLY(8) // Percent<256
237 TEST_AND_APPLY(7) // Percent<512
238 TEST_AND_APPLY(6) // Percent<1024
239
240 #undef TEST_AND_APPLY
241
242 // Percent<2048
243 return _percentageApprox<5U>(percent, value);
244}

320static inline uint16_t udiv_32_16 (uint32_t dividend, uint16_t divisor)
321{
322#if defined(__AVR__)
323
324 if (divisor==0U) { return UINT16_MAX; }
325
326 #define INDEX_REG "r16"
327
328 asm(
329 " ldi " INDEX_REG ", 16 ; bits = 16\n\t"
330 "0:\n\t"
331 " lsl %A0 ; shift\n\t"
332 " rol %B0 ; rem:quot\n\t"
333 " rol %C0 ; left\n\t"
334 " rol %D0 ; by 1\n\t"
335 " brcs 1f ; if carry out, rem > divisor\n\t"
336 " cp %C0, %A1 ; is rem less\n\t"
337 " cpc %D0, %B1 ; than divisor ?\n\t"
338 " brcs 2f ; yes, when carry out\n\t"
339 "1:\n\t"
340 " sub %C0, %A1 ; compute\n\t"
341 " sbc %D0, %B1 ; rem -= divisor\n\t"
342 " ori %A0, 1 ; record quotient bit as 1\n\t"
343 "2:\n\t"
344 " dec " INDEX_REG " ; bits--\n\t"
345 " brne 0b ; until bits == 0"
346 : "=d" (dividend)
347 : "d" (divisor) , "0" (dividend)
348 : INDEX_REG
349 );
350
351 // Lower word contains the quotient, upper word contains the remainder.
352 return dividend & 0xFFFFU;
353#else
354 // The non-AVR platforms are all fast enough (or have built in hardware dividers)
355 // so just fall back to regular 32-bit division.
356 return dividend / divisor;
357#endif
358}

416template <typename T, typename TPrime>
417static inline T LOW_PASS_FILTER_8BIT(T input, uint8_t alpha, T prior) {
418 // Intermediate steps are for MISRA compliance
419 // Equivalent to: (input * (256 - alpha) + (prior * alpha)) >> 8
420 static constexpr T ALPHA_MAX = (T)256;
421 T inv_alpha = ALPHA_MAX - (T)alpha;
422 TPrime prior_alpha = (prior * (TPrime)alpha);
423 TPrime preshift = (input * (TPrime)inv_alpha) + prior_alpha;
424 return (T)(preshift / ALPHA_MAX); // Division should resolve to a shift & avoids a MISRA violation
425}
