215static inline uint32_t percentageApprox(uint16_t percent, uint32_t value) {
216 // To keep the percentage within 16-bits (for performance), we have to scale the precision based on the percentage.
217 // I.e. the larger the percentage, the smaller the precision has to be (and vice-versa).
218 //
219 // We could use __builtin_clz() and use the leading zero count as the precision, but that is slow:
220 // * AVR doesn't have a clz ASM instruction, so __builtin_clz() is implemented in software
221 // * It would require removing some compile time optimizations
222 #define TEST_AND_APPLY(precision) \
223 if (percent<(UINT16_C(1)<<(UINT16_C(16)-(precision)))) { \
224 return _percentageApprox<(precision)>(percent, value); \
225 }
226
227 TEST_AND_APPLY(9) // Percent<128
228 TEST_AND_APPLY(8) // Percent<256
229 TEST_AND_APPLY(7) // Percent<512
230 TEST_AND_APPLY(6) // Percent<1024
231
232 #undef TEST_AND_APPLY
233
234 // Percent<2048
235 return _percentageApprox<5U>(percent, value);
236}

311static inline uint16_t udiv_32_16 (uint32_t dividend, uint16_t divisor)
312{
313#if defined(__AVR__)
314
315 if (divisor==0U) { return UINT16_MAX; }
316
317 #define INDEX_REG "r16"
318
319 asm(
320 " ldi " INDEX_REG ", 16 ; bits = 16\n\t"
321 "0:\n\t"
322 " lsl %A0 ; shift\n\t"
323 " rol %B0 ; rem:quot\n\t"
324 " rol %C0 ; left\n\t"
325 " rol %D0 ; by 1\n\t"
326 " brcs 1f ; if carry out, rem > divisor\n\t"
327 " cp %C0, %A1 ; is rem less\n\t"
328 " cpc %D0, %B1 ; than divisor ?\n\t"
329 " brcs 2f ; yes, when carry out\n\t"
330 "1:\n\t"
331 " sub %C0, %A1 ; compute\n\t"
332 " sbc %D0, %B1 ; rem -= divisor\n\t"
333 " ori %A0, 1 ; record quotient bit as 1\n\t"
334 "2:\n\t"
335 " dec " INDEX_REG " ; bits--\n\t"
336 " brne 0b ; until bits == 0"
337 : "=d" (dividend)
338 : "d" (divisor) , "0" (dividend)
339 : INDEX_REG
340 );
341
342 // Lower word contains the quotient, upper word contains the remainder.
343 return dividend & 0xFFFFU;
344#else
345 // The non-AVR platforms are all fast enough (or have built in hardware dividers)
346 // so just fall back to regular 32-bit division.
347 return dividend / divisor;
348#endif
349}

407template <typename T, typename TPrime>
408static inline T LOW_PASS_FILTER_8BIT(T input, uint8_t alpha, T prior) {
409 // Intermediate steps are for MISRA compliance
410 // Equivalent to: (input * (256 - alpha) + (prior * alpha)) >> 8
411 static constexpr T ALPHA_MAX = (T)256;
412 T inv_alpha = ALPHA_MAX - (T)alpha;
413 TPrime prior_alpha = (prior * (TPrime)alpha);
414 TPrime preshift = (input * (TPrime)inv_alpha) + prior_alpha;
415 return (T)(preshift / ALPHA_MAX); // Division should resolve to a shift & avoids a MISRA violation
416}

469static inline uint8_t fast_map(const uint8_t from, const uint8_t fromLow, const uint8_t fromHigh, const uint8_t toLow, const uint8_t toHigh) {
470 // Stick to unsigned math for performance, so need to check for output range inversion
471 if (toLow>toHigh) {
472 return toLow - scale(from - fromLow, fromHigh - fromLow, toLow-toHigh);
