242static inline uint32_t percentageApprox(uint16_t percent, uint32_t value) {
243 // To keep the percentage within 16-bits (for performance), we have to scale the precision based on the percentage.
244 // I.e. the larger the percentage, the smaller the precision has to be (and vice-versa).
245 //
246 // We could use __builtin_clz() and use the leading zero count as the precision, but that is slow:
247 // * AVR doesn't have a clz ASM instruction, so __builtin_clz() is implemented in software
248 // * It would require removing some compile time optimizations
249 #define TEST_AND_APPLY(precision) \
250 if (percent<(UINT16_C(1)<<(UINT16_C(16)-(precision)))) { \
251 return _percentageApprox<(precision)>(percent, value); \
252 }
253
254 TEST_AND_APPLY(9) // Percent<128
255 TEST_AND_APPLY(8) // Percent<256
256 TEST_AND_APPLY(7) // Percent<512
257 TEST_AND_APPLY(6) // Percent<1024
258
259 #undef TEST_AND_APPLY
260
261 // Percent<2048
262 return _percentageApprox<5U>(percent, value);
263}

356template <typename T, typename TPrime>
357static inline T LOW_PASS_FILTER_8BIT(T input, uint8_t alpha, T prior) {
358 // Intermediate steps are for MISRA compliance
359 // Equivalent to: (input * (256 - alpha) + (prior * alpha)) >> 8
360 static constexpr uint16_t ALPHA_MAX_SHIFT = 8U;
361 static constexpr uint16_t ALPHA_MAX = 2U << (ALPHA_MAX_SHIFT-1U);
362 uint16_t inv_alpha = ALPHA_MAX - alpha;
363 TPrime prior_alpha = (prior * (TPrime)alpha);
364 TPrime preshift = (input * (TPrime)inv_alpha) + prior_alpha;
365 return (T)(preshift >> (TPrime)ALPHA_MAX_SHIFT);
366}

419static inline uint8_t fast_map(const uint8_t from, const uint8_t fromLow, const uint8_t fromHigh, const uint8_t toLow, const uint8_t toHigh) {
420 // Stick to unsigned math for performance, so need to check for output range inversion
421 if (toLow>toHigh) {
422 return toLow - scale(from - fromLow, fromHigh - fromLow, toLow-toHigh);
423 } else {
424 return scale(from - fromLow, fromHigh - fromLow, toHigh-toLow) + toLow;
