250static inline uint16_t udiv_32_16 (uint32_t dividend, uint16_t divisor)
251{
252#if defined(CORE_AVR) || defined(ARDUINO_ARCH_AVR)
253
254 if (divisor==0U || !udiv_is16bit_result(dividend, divisor)) { return UINT16_MAX; }
255
256 #define INDEX_REG "r16"
257
258 asm(
259 " ldi " INDEX_REG ", 16 ; bits = 16\n\t"
260 "0:\n\t"
261 " lsl %A0 ; shift\n\t"
262 " rol %B0 ; rem:quot\n\t"
263 " rol %C0 ; left\n\t"
264 " rol %D0 ; by 1\n\t"
265 " brcs 1f ; if carry out, rem > divisor\n\t"
266 " cp %C0, %A1 ; is rem less\n\t"
267 " cpc %D0, %B1 ; than divisor ?\n\t"
268 " brcs 2f ; yes, when carry out\n\t"
269 "1:\n\t"
270 " sub %C0, %A1 ; compute\n\t"
271 " sbc %D0, %B1 ; rem -= divisor\n\t"
272 " ori %A0, 1 ; record quotient bit as 1\n\t"
273 "2:\n\t"
274 " dec " INDEX_REG " ; bits--\n\t"
275 " brne 0b ; until bits == 0"
276 : "=d" (dividend)
277 : "d" (divisor) , "0" (dividend)
278 : INDEX_REG
279 );
280
281 // Lower word contains the quotient, upper word contains the remainder.
282 return dividend & 0xFFFFU;
283#else
284 // The non-AVR platforms are all fast enough (or have built in hardware dividers)
285 // so just fall back to regular 32-bit division.
286 return dividend / divisor;
287#endif
288}
