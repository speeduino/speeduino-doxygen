16template <typename axis_t, typename value_t>
17struct Table2DCache
18{
19 // Store the upper index of the bin we last found. This is used to make the next check faster
20 // Since this is the *upper* index, it can never be 0.

22
23 //Store the last input and output for caching
24 axis_t lastInput = 0;

26 uint8_t cacheTime = 0U; //Tracks when the last cache value was set so it can expire after x seconds. A timeout is required to pickup when a tuning value is changed, otherwise the old cached value will continue to be returned as the X value isn't changing.

106template <typename T, uint8_t sizeT>
107static inline Bin<T> findBin(const T *const array, const T value)
108{
109 // Loop from the upper end of the axis back down to the 1st bin [0,1]
110 // Assume array is ordered [min...max]
111 const T *binLower = array+sizeT-2U;
112 while ( (*binLower >= value) && (binLower != array) ) { --binLower; }
113 return Bin<T>(binLower-array+1U, *(binLower+1U), *binLower);
114}

140template <typename axis_t, typename value_t, uint8_t sizeT>
141static inline value_t table2D_getValue(const table2D<axis_t, value_t, sizeT> *fromTable, const axis_t axisValue)
142{
143 //Check whether the X input is the same as last time this ran
144 if( (axisValue == fromTable->cache.lastInput) && (!cacheExpired(fromTable->cache)) ) { return fromTable->cache.lastOutput; }
145 else if(axisValue >= fromTable->axis[sizeT-1]) // Test if above the max axis value, clip to max data value
146 {
147 fromTable->cache.lastOutput = fromTable->values[sizeT-1];
148 fromTable->cache.lastBinUpperIndex = sizeT-1;
149 }
150 else if (axisValue <= fromTable->axis[0]) // Test if below the min axis value, clip to min data value
151 {
152 fromTable->cache.lastOutput = fromTable->values[0];
153 fromTable->cache.lastBinUpperIndex = 1U;
154 }
155 else
156 {
157 // None of the cached or last values match, so we need to find the new value
158 // 1st check is whether we're still in the same X bin as last time
159 _table2d_detail::Bin<axis_t> xBin = _table2d_detail::Bin<axis_t>(fromTable->axis, fromTable->cache.lastBinUpperIndex);
160 if (!xBin.withinBin(axisValue))
161 {
162 //If we're not in the same bin, search
163 xBin = _table2d_detail::findBin<axis_t, sizeT>(fromTable->axis, axisValue);
164 }
165
166 // We are exactly at the bin upper bound, so no need to interpolate
167 if (axisValue==xBin.upperValue())
168 {
169 fromTable->cache.lastOutput = fromTable->values[xBin.upperIndex];
170 fromTable->cache.lastBinUpperIndex = xBin.upperIndex;
171 }
172 else // Must be within the bin, interpolate
173 {
174 fromTable->cache.lastOutput = _table2d_detail::interpolate(axisValue, xBin, _table2d_detail::Bin<value_t>(fromTable->values, xBin.upperIndex));
175 fromTable->cache.lastBinUpperIndex = xBin.upperIndex;
176 }
177 // Note: we cannot be at the bin lower bound here, as that would violate the bin definition of a non-inclusive lower bound
178 }
179 fromTable->cache.cacheTime = _table2d_detail::getCacheTime(); //As we're not using the cache value, set the current secl value to track when this new value was calculated
180 fromTable->cache.lastInput = axisValue;
181
182 return fromTable->cache.lastOutput;
183}
