190static inline bool isRunning(const Schedule &schedule) {
191 // Using flags and bitwise AND (&) to check multiple states is much quicker
192 // than a logical or (||) (one less branch & 30% less instructions)
193 static constexpr uint8_t flags = RUNNING | RUNNING_WITHNEXT;
194 return ((uint8_t)schedule.Status & flags)!=0U;
195}
