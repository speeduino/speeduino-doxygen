142static inline bool isRunning(const Schedule &schedule) {
143 // Using flags and bitwise AND (&) to check multiple states is much quicker
144 // than a logical or (||) (one less branch & 30% less instructions)
145 static constexpr uint8_t flags = RUNNING | RUNNING_WITHNEXT;
146 return ((uint8_t)schedule.Status & flags)!=0U;
147}
