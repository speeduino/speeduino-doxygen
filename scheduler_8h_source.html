1 /*
2 This scheduler is designed to maintain 2 schedules for use by the fuel and ignition systems.
3 It functions by waiting for the overflow vectors from each of the timers in use to overflow, which triggers an interrupt
4 
5 /Technical
6 Currently I am prescaling the 16-bit timers to 256 for injection and 64 for ignition. This means that the counter increments every 16us (injection) / 4uS (ignition) and will overflow every 1048576uS
7 Max Period = (Prescale)*(1/Frequency)*(2^17)
8 (See playground.arduino.cc/code/timer1)
9 This means that the precision of the scheduler is 16uS (+/- 8uS of target) for fuel and 4uS (+/- 2uS) for ignition
10 
11 /Features
12 This differs from most other schedulers in that its calls are non-recurring (IE You schedule an event at a certain time and once it has occurred, it will not reoccur unless you explicitely ask for it)
13 Each timer can have only 1 callback associated with it at any given time. If you call the setCallback function a 2nd time, the original schedule will be overwritten and not occur
14 
15 Timer identification
16 The Arduino timer3 is used for schedule 1
17 The Arduino timer4 is used for schedule 2
18 Both of these are 16-bit timers (ie count to 65536)
19 See page 136 of the processors datasheet: www.atmel.com/Images/doc2549.pdf
20 
21 256 prescale gives tick every 16uS
22 256 prescale gives overflow every 1048576uS (This means maximum wait time is 1.0485 seconds)
23 
24 */

151 static inline unsigned int setQueue(volatile Schedule *queue[], Schedule *schedule1, Schedule *schedule2, unsigned int CNT)
152 {
153  //Create an array of all the upcoming targets, relative to the current count on the timer
154  unsigned int tmpQueue[4];
155 
156  //Set the initial queue state. This order matches the tmpQueue order
157  if(schedule1->Status == OFF)
158  {
159  queue[0] = schedule2;
160  queue[1] = schedule2;
161  tmpQueue[0] = schedule2->startCompare - CNT;
162  tmpQueue[1] = schedule2->endCompare - CNT;
163  }
164  else
165  {
166  queue[0] = schedule1;
167  queue[1] = schedule1;
168  tmpQueue[0] = schedule1->startCompare - CNT;
169  tmpQueue[1] = schedule1->endCompare - CNT;
170  }
171 
172  if(schedule2->Status == OFF)
173  {
174  queue[2] = schedule1;
175  queue[3] = schedule1;
176  tmpQueue[2] = schedule1->startCompare - CNT;
177  tmpQueue[3] = schedule1->endCompare - CNT;
178  }
179  else
180  {
181  queue[2] = schedule2;
182  queue[3] = schedule2;
183  tmpQueue[2] = schedule2->startCompare - CNT;
184  tmpQueue[3] = schedule2->endCompare - CNT;
185  }
186 
187 
188  //Sort the queues. Both queues are kept in sync.
189  //This implementes a sorting networking based on the Bose-Nelson sorting network
190  //See: pages.ripco.net/~jgamble/nw.html
191  #define SWAP(x,y) if(tmpQueue[y] < tmpQueue[x]) { unsigned int tmp = tmpQueue[x]; tmpQueue[x] = tmpQueue[y]; tmpQueue[y] = tmp; volatile Schedule *tmpS = queue[x]; queue[x] = queue[y]; queue[y] = tmpS; }
192  /*SWAP(0, 1); */ //Likely not needed
193  /*SWAP(2, 3); */ //Likely not needed
194  SWAP(0, 2);
195  SWAP(1, 3);
196  SWAP(1, 2);
197 
198  //Return the next compare time in the queue
199  return tmpQueue[0] + CNT; //Return the
200 }
