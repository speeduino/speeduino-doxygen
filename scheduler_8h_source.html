219static inline COMPARE_TYPE setQueue(volatile Schedule *queue[], Schedule *schedule1, Schedule *schedule2, unsigned int CNT)
220{
221 //Create an array of all the upcoming targets, relative to the current count on the timer
222 unsigned int tmpQueue[4];
223
224 //Set the initial queue state. This order matches the tmpQueue order
225 if(schedule1->Status == OFF)
226 {
227 queue[0] = schedule2;
228 queue[1] = schedule2;
229 tmpQueue[0] = schedule2->startCompare - CNT;
230 tmpQueue[1] = schedule2->endCompare - CNT;
231 }
232 else
233 {
234 queue[0] = schedule1;
235 queue[1] = schedule1;
236 tmpQueue[0] = schedule1->startCompare - CNT;
237 tmpQueue[1] = schedule1->endCompare - CNT;
238 }
239
240 if(schedule2->Status == OFF)
241 {
242 queue[2] = schedule1;
243 queue[3] = schedule1;
244 tmpQueue[2] = schedule1->startCompare - CNT;
245 tmpQueue[3] = schedule1->endCompare - CNT;
246 }
247 else
248 {
249 queue[2] = schedule2;
250 queue[3] = schedule2;
251 tmpQueue[2] = schedule2->startCompare - CNT;
252 tmpQueue[3] = schedule2->endCompare - CNT;
253 }
254
255

258 //See: pages.ripco.net/~jgamble/nw.html

270/*
271 * Moves all the Schedules in a queue forward one position.
272 * The current item (0) is discarded
273 * The final queue slot is set to nullSchedule to indicate that no action should be taken

275static inline unsigned int popQueue(volatile Schedule *queue[])
276{
277 queue[0] = queue[1];
278 queue[1] = queue[2];
279 queue[2] = queue[3];
280 //queue[3] = &nullSchedule;
281
282 unsigned int returnCompare;
283 if( queue[0]->Status == PENDING ) { returnCompare = queue[0]->startCompare; }
284 else { returnCompare = queue[0]->endCompare; }
285
286 return returnCompare;
287}
